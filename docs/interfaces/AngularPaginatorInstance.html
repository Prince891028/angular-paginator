export interface AngularPaginatorInstance {
    /** id of the pagination instance */
    id?: string;
    /** total numeber of items to be paginated. usually array length */
    totalItems: number;
    /** current page number */
    currentPage: number;
    /** total items to be shown per page */
    itemsPerPage: number;
}

    active: boolean;
}
